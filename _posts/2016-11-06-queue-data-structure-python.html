Below is a basic Python implementation of a queue - a FIFO data structure:

{% highlight python %} class Queue(object): """ An implementation of a queue - a FIFO data structure FIFO - 'First In First Out' """ def __init__(self, queue=None): """ Construct a new queue Args: queue - a list of elements (optional) """ if queue is None: self.queue = [] else: self.queue = list(queue) def dequeue(self): """ Deqeue an element from the queue Returns: Return the first element that was added to the queue. """ return self.queue.pop(0) def enqueue(self, element): """ Enqeue an element to the end of the queue Args: element - the element to enqueue """ self.queue.append(element) def isEmpty(self): """ Check if the queue is empty Returns: Return True if the queue is empty or False if it is not """ return len(self.queue) == 0 {% endhighlight %}
